CREATE TABLE Projet
(
	idProjet INT PRIMARY KEY NOT NULL,
	nomProjet VARCHAR(20) NOT NULL,
	objetProt VARCHAR(100) NOT NULL,
	comProjet VARCHAR(100) NOT NULL,
	nComProjet INT NOT NULL,
	dateReception DATETIME NOT NULL,
	designationProjet VARCHAR(50) NOT NULL,
	idclient int references Client(idClient)
)

CREATE TABLE Client
(
	idClient INT PRIMARY KEY NOT NULL,
	nomClient VARCHAR(20),
	telClient VARCHAR(20),
	societe VARCHAR(20),
	adresse1 VARCHAR(50),
	adresse2 VARCHAR(50),
	ville VARCHAR(20)
)


CREATE TABLE Secteur
(
	idSecteur INT PRIMARY KEY NOT NULL,
	nomSecteur VARCHAR(20) NOT NULL,
	siteSecteur VARCHAR(50),
	adresse1 VARCHAR(50),
	adresse2 VARCHAR(50),
	prestation VARCHAR(50),
	expediteur VARCHAR(50),
	destinataire VARCHAR(50),
	recuLe VARCHAR(20),
	recuPar VARCHAR(20),
	retourLe VARCHAR(20),
	retourPar VARCHAR(20),
	idprojet INT REFERENCES Projet(idProjet)
)

CREATE TABLE Prestation
(
	idPrestation INT PRIMARY KEY NOT NULL,
	refPrestation VARCHAR(20) UNIQUE NOT NULL,
	nomPrestation VARCHAR(20) NOT NULL,
	idProjet INT REFERENCES Projet(idProjet)
)

CREATE TABLE Utilisateur
(
